<html>
    <head>
        <body>
            <script>
function bookFlight() {
    return new Promise((resolve, reject) => {
      // Simulating asynchronous flight booking
      setTimeout(() => {
        resolve('Flight booked successfully');
      }, 2000);
    });
  }
  
  function bookHotel() {
    return new Promise((resolve, reject) => {
      // Simulating asynchronous hotel booking
      setTimeout(() => {
        resolve('Hotel booked successfully');
      }, 1500);
    });
  }
  
  bookFlight()
    .then(() => bookHotel())
    .then((message) => {
      console.log(message);
    })
    .catch((error) => {
      console.log(error);
    });
</script>
        </body>
    </head>
</html>
